using Microsoft.EntityFrameworkCore;
using Yalla.Application.DTO.Request;
using Yalla.Application.Services;
using Yalla.Application.UnitTests.TestInfrastructure;
using Yalla.Domain.Entities;
using Yalla.Domain.Enums;

namespace Yalla.Application.UnitTests.Services;

public class ClientServiceTests
{
  [Fact]
  public async Task RegisterClientAsync_CreatesClient()
  {
    using var scope = TestDbFactory.Create();
    var service = new ClientService(scope.Db);

    var response = await service.RegisterClientAsync(new RegisterClientRequest
    {
      Name = "Client",
      PhoneNumber = "992300001"
    });

    Assert.Equal("Client", response.Client.Name);
    Assert.Single(scope.Db.Clients);
  }

  [Fact]
  public async Task RegisterClientAsync_ThrowsForDuplicatePhone()
  {
    using var scope = TestDbFactory.Create();
    var existing = TestDbFactory.CreateUser("Any", "992300002", Role.SuperAdmin);
    scope.Db.Users.Add(existing);
    await scope.Db.SaveChangesAsync();

    var service = new ClientService(scope.Db);

    await Assert.ThrowsAsync<InvalidOperationException>(() => service.RegisterClientAsync(new RegisterClientRequest
    {
      Name = "Client",
      PhoneNumber = "992300002"
    }));
  }

  [Fact]
  public async Task AddProductToBasketAsync_UpdatesExistingPositionQuantity()
  {
    using var scope = TestDbFactory.Create();
    var client = TestDbFactory.CreateClient("C", "992300003");
    var medicine = TestDbFactory.CreateMedicine("M", "M-1");
    scope.Db.Clients.Add(client);
    scope.Db.Medicines.Add(medicine);
    await scope.Db.SaveChangesAsync();

    var existing = new BasketPosition(client.Id, medicine.Id, medicine, 1);
    scope.Db.BasketPositions.Add(existing);
    await scope.Db.SaveChangesAsync();

    var service = new ClientService(scope.Db);
    var response = await service.AddProductToBasketAsync(new AddProductToBasketRequest
    {
      ClientId = client.Id,
      MedicineId = medicine.Id,
      Quantity = 2
    });

    Assert.Equal(3, response.BasketPosition.Quantity);
  }

  [Fact]
  public async Task UpdateBasketPositionQuantityAsync_ThrowsForNonPositiveQuantity()
  {
    using var scope = TestDbFactory.Create();
    var service = new ClientService(scope.Db);

    await Assert.ThrowsAsync<Yalla.Domain.Exceptions.DomainArgumentException>(() =>
      service.UpdateBasketPositionQuantityAsync(new UpdateBasketPositionQuantityRequest
      {
        ClientId = Guid.NewGuid(),
        PositionId = Guid.NewGuid(),
        Quantity = 0
      }));
  }

  [Fact]
  public async Task CheckoutBasketAsync_Success_DecreasesStockAndMovesToUnderReview()
  {
    using var scope = TestDbFactory.Create();
    var client = TestDbFactory.CreateClient("C", "992300004");
    var pharmacy = TestDbFactory.CreatePharmacy("P", "Addr", Guid.NewGuid());
    var medicine = TestDbFactory.CreateMedicine("M", "M-2");
    scope.Db.Clients.Add(client);
    scope.Db.Pharmacies.Add(pharmacy);
    scope.Db.Medicines.Add(medicine);
    await scope.Db.SaveChangesAsync();

    var offer = TestDbFactory.CreateOffer(medicine.Id, pharmacy.Id, stock: 10, price: 5m);
    scope.Db.Offers.Add(offer);

    var basket = new BasketPosition(client.Id, medicine.Id, medicine, 3);
    scope.Db.BasketPositions.Add(basket);
    await scope.Db.SaveChangesAsync();

    var service = new ClientService(scope.Db);
    var response = await service.CheckoutBasketAsync(new CheckoutBasketRequest
    {
      ClientId = client.Id,
      PharmacyId = pharmacy.Id,
      DeliveryAddress = "Home"
    });

    scope.Db.ChangeTracker.Clear();
    var savedOffer = await scope.Db.Offers.AsNoTracking().FirstAsync(x => x.Id == offer.Id);
    var order = await scope.Db.Orders.FirstAsync(x => x.Id == response.OrderId);
    var basketCount = await scope.Db.BasketPositions.CountAsync(x => x.ClientId == client.Id);

    Assert.Equal(7, savedOffer.StockQuantity);
    Assert.Equal(Status.UnderReview, order.Status);
    Assert.Equal(0, basketCount);
  }

  [Fact]
  public async Task CheckoutBasketAsync_IgnoredPosition_RemainsInBasketAndMarkedRejectedInOrder()
  {
    using var scope = TestDbFactory.Create();
    var client = TestDbFactory.CreateClient("C", "992300005");
    var pharmacy = TestDbFactory.CreatePharmacy("P", "Addr", Guid.NewGuid());
    var med1 = TestDbFactory.CreateMedicine("M1", "M-3");
    var med2 = TestDbFactory.CreateMedicine("M2", "M-4");
    scope.Db.Clients.Add(client);
    scope.Db.Pharmacies.Add(pharmacy);
    scope.Db.Medicines.AddRange(med1, med2);
    await scope.Db.SaveChangesAsync();

    scope.Db.Offers.AddRange(
      TestDbFactory.CreateOffer(med1.Id, pharmacy.Id, 10, 5m),
      TestDbFactory.CreateOffer(med2.Id, pharmacy.Id, 10, 7m));

    var p1 = new BasketPosition(client.Id, med1.Id, med1, 1);
    var p2 = new BasketPosition(client.Id, med2.Id, med2, 1);
    scope.Db.BasketPositions.AddRange(p1, p2);
    await scope.Db.SaveChangesAsync();

    var service = new ClientService(scope.Db);
    var checkout = await service.CheckoutBasketAsync(new CheckoutBasketRequest
    {
      ClientId = client.Id,
      PharmacyId = pharmacy.Id,
      DeliveryAddress = "Home",
      IgnoredPositionIds = [p2.Id]
    });

    var orderPositions = await scope.Db.OrderPositions
      .Where(x => x.OrderId == checkout.OrderId)
      .ToListAsync();
    var remainingBasket = await scope.Db.BasketPositions
      .Where(x => x.ClientId == client.Id)
      .ToListAsync();

    Assert.Single(remainingBasket);
    Assert.Equal(p2.Id, remainingBasket.Single().Id);
    Assert.Contains(orderPositions, x => x.MedicineId == med2.Id && x.IsRejected);
    Assert.Contains(orderPositions, x => x.MedicineId == med1.Id && !x.IsRejected);
  }

  [Fact]

    var medicine = TestDbFactory.CreateMedicine("M", "M-5");

    scope.Db.Offers.Add(TestDbFactory.CreateOffer(medicine.Id, pharmacy.Id, 10, 5m));

    await Assert.ThrowsAsync<InvalidOperationException>(() => service.CheckoutBasketAsync(new CheckoutBasketRequest

      PharmacyId = pharmacy.Id,

  }

  [Fact]
  public async Task ClearBasketAsync_RemovesAllPositions()
  {
    using var scope = TestDbFactory.Create();
    var client = TestDbFactory.CreateClient("C", "992300007");
    var medicine = TestDbFactory.CreateMedicine("M", "M-6");
    scope.Db.Clients.Add(client);
    scope.Db.Medicines.Add(medicine);
    await scope.Db.SaveChangesAsync();

    scope.Db.BasketPositions.AddRange(
      new BasketPosition(client.Id, medicine.Id, medicine, 1),
      new BasketPosition(client.Id, medicine.Id, medicine, 2));
    await scope.Db.SaveChangesAsync();

    var service = new ClientService(scope.Db);
    var response = await service.ClearBasketAsync(new ClearBasketRequest { ClientId = client.Id });

    Assert.Equal(0, response.BasketItemsCount);
    Assert.Empty(await scope.Db.BasketPositions.Where(x => x.ClientId == client.Id).ToListAsync());
  }
