  [Fact]
  public async Task PreviewCheckoutAsync_ReturnsReasonsAndTotals()
  {
    using var scope = TestDbFactory.Create();
    var client = TestDbFactory.CreateClient("C", "994100028");
    var pharmacy = TestDbFactory.CreatePharmacy("P", "Addr", Guid.NewGuid(), isActive: false);
    var availableMed = TestDbFactory.CreateMedicine("Available", "MED-10");
    var lowStockMed = TestDbFactory.CreateMedicine("LowStock", "MED-11");
    scope.Db.Clients.Add(client);
    scope.Db.Pharmacies.Add(pharmacy);
    scope.Db.Medicines.AddRange(availableMed, lowStockMed);
    await scope.Db.SaveChangesAsync();

    var acceptedPosition = new BasketPosition(client.Id, availableMed.Id, availableMed, 2);
    var rejectedPosition = new BasketPosition(client.Id, lowStockMed.Id, lowStockMed, 5);
    scope.Db.BasketPositions.AddRange(acceptedPosition, rejectedPosition);
    scope.Db.Offers.AddRange(
      TestDbFactory.CreateOffer(availableMed.Id, pharmacy.Id, 10, 3m),
      TestDbFactory.CreateOffer(lowStockMed.Id, pharmacy.Id, 1, 4m));
    await scope.Db.SaveChangesAsync();

    var service = new ClientService(scope.Db);
    var response = await service.PreviewCheckoutAsync(new CheckoutBasketRequest
    {
      ClientId = client.Id,
      PharmacyId = pharmacy.Id,
      DeliveryAddress = "Home"
    });

    Assert.False(response.CanCheckout);
    Assert.Equal(2, response.TotalPositions);
    Assert.Equal(1, response.AcceptedPositionsCount);
    Assert.Equal(1, response.RejectedPositionsCount);
    Assert.Equal(6m, response.Cost);
    Assert.Equal(20m, response.ReturnCost);
    Assert.Contains(response.Positions, x => x.PositionId == rejectedPosition.Id && x.Reason == "InsufficientStock");
  }

  private static async Task<(Client Client, Pharmacy Pharmacy, Medicine Medicine, BasketPosition BasketPosition)> SeedCheckoutBaseAsync(
    TestDbScope scope,
    string phone,
    bool isMedicineActive,
    bool createOffer,
    int stock)
  {
    var client = TestDbFactory.CreateClient("Client", phone);
    var pharmacy = TestDbFactory.CreatePharmacy("P", "Addr", Guid.NewGuid());
    var medicine = TestDbFactory.CreateMedicine("M", $"ART-{phone}", isActive: isMedicineActive);
    scope.Db.Clients.Add(client);
    scope.Db.Pharmacies.Add(pharmacy);
    scope.Db.Medicines.Add(medicine);
    await scope.Db.SaveChangesAsync();

    if (createOffer)
      scope.Db.Offers.Add(TestDbFactory.CreateOffer(medicine.Id, pharmacy.Id, stock, 5m));

    var basketPosition = new BasketPosition(client.Id, medicine.Id, medicine, 2);
    scope.Db.BasketPositions.Add(basketPosition);
    await scope.Db.SaveChangesAsync();

    return (client, pharmacy, medicine, basketPosition);
  }
