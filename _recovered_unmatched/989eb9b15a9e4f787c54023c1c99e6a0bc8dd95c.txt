  [Fact]
  public async Task GetClientOrderHistoryAsync_ReturnsOrdersSortedByPlacedAtDesc()
  {
    using var scope = TestDbFactory.Create();
    var client = TestDbFactory.CreateClient("Client", "995100001");
    var pharmacy = TestDbFactory.CreatePharmacy("P", "Addr", Guid.NewGuid());
    var medicine = TestDbFactory.CreateMedicine("M", "ORD-1");
    scope.Db.Clients.Add(client);
    scope.Db.Pharmacies.Add(pharmacy);
    scope.Db.Medicines.Add(medicine);
    await scope.Db.SaveChangesAsync();

    var oldOrderId = Guid.NewGuid();
    var oldOrder = new Order(
      oldOrderId,
      client.Id,
      pharmacy.Id,
      "Addr",
      [
        new OrderPosition(oldOrderId, medicine.Id, medicine, new OfferSnapshot(pharmacy.Id, 10m), 1)
      ],
      new DateTime(2026, 1, 1, 10, 0, 0));

    var newOrderId = Guid.NewGuid();
    var newOrder = new Order(
      newOrderId,
      client.Id,
      pharmacy.Id,
      "Addr",
      [
        new OrderPosition(newOrderId, medicine.Id, medicine, new OfferSnapshot(pharmacy.Id, 20m), 2)
      ],
      new DateTime(2026, 1, 2, 10, 0, 0));

    scope.Db.Orders.AddRange(oldOrder, newOrder);
    await scope.Db.SaveChangesAsync();

    var service = new OrderService(scope.Db);
    var response = await service.GetClientOrderHistoryAsync(new GetClientOrderHistoryRequest { ClientId = client.Id });

    Assert.Equal(2, response.Orders.Count);
    Assert.Equal(newOrder.Id, response.Orders.First().OrderId);
  }

  private static async Task<(Client Client, Pharmacy Pharmacy, PharmacyWorker Worker, Medicine Medicine)> SeedWorkerSetupAsync(
    TestDbScope scope,
    string basePhone)
  {
    var client = TestDbFactory.CreateClient("Client", basePhone);
    var pharmacy = TestDbFactory.CreatePharmacy("Ph", "Addr", Guid.NewGuid());
    var worker = TestDbFactory.CreateWorker("Worker", $"{basePhone}9", pharmacy.Id, pharmacy);
    var medicine = TestDbFactory.CreateMedicine("M", $"ART-{basePhone}");
    scope.Db.Clients.Add(client);
    scope.Db.Pharmacies.Add(pharmacy);
    scope.Db.PharmacyWorkers.Add(worker);
    scope.Db.Medicines.Add(medicine);
    await scope.Db.SaveChangesAsync();

    return (client, pharmacy, worker, medicine);
  }

  private static async Task<Order> CreateOrderWithStatus(
    TestDbScope scope,
    Guid clientId,
    Guid pharmacyId,
    Medicine medicine,
    Status status)
  {
    var order = TestDbFactory.CreateOrder(clientId, pharmacyId, "Address", (medicine, 10m, 1, false));

    while (order.Status != status)
    {
      order.NextStage(true);
      if ((int)order.Status > (int)status)
        throw new InvalidOperationException("Cannot reach requested status with NextStage.");
    }

    scope.Db.Orders.Add(order);
    await scope.Db.SaveChangesAsync();
    return order;
  }
